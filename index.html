#include #include #include #include #include "soc/soc.h" // ブラウンアウト対策用 #include "soc/rtc_cntl_reg.h" // ブラウンアウト対策用 // --- 設定値 --- #define SERVICE_UUID "4fafc201-1fb5-459e-8fcc-c5c9c331914b" #define CHARACTERISTIC_UUID "beb5483e-36e1-4688-b7f5-ea07361b26a8" // ============================================================================= // ピン設定 (DRV8835 IN/INモード想定) // ============================================================================= const int motorA_IN1 = 27; // 左モーター前進用 const int motorA_IN2 = 32; // 左モーター後退用 const int motorB_IN1 = 25; // 右モーター前進用 const int motorB_IN2 = 26; // 右モーター後退用 const int PWM_FREQ = 20000; const int PWM_RESOLUTION = 8; // 0-255 BLEServer* pServer = NULL; BLECharacteristic* pCharacteristic = NULL; bool deviceConnected = false; bool oldDeviceConnected = false; // フェールセーフ用変数 volatile unsigned long lastCommunicationTime = 0; const unsigned long FAILSAFE_TIMEOUT = 1000; // 1秒通信途絶で目標値ゼロ // --- モーター制御用変数 --- // Webアプリから受信した「目標」のPWM値 (-255 〜 255) int targetLeftPWM = 0; int targetRightPWM = 0; // 現在「実際」に出力しているPWM値 (ソフトスタート用) float currentLeftPWM = 0; float currentRightPWM = 0; // モーター出力関数 (内部で呼ばれます) void setMotors(int leftPWM, int rightPWM) { // --- 左モーター (A) --- if (leftPWM > 0) { ledcWrite(motorA_IN1, leftPWM); ledcWrite(motorA_IN2, 0); } else if (leftPWM < 0) { ledcWrite(motorA_IN1, 0); ledcWrite(motorA_IN2, abs(leftPWM)); } else { ledcWrite(motorA_IN1, 0); ledcWrite(motorA_IN2, 0); } // --- 右モーター (B) --- if (rightPWM > 0) { ledcWrite(motorB_IN1, rightPWM); ledcWrite(motorB_IN2, 0); } else if (rightPWM < 0) { ledcWrite(motorB_IN1, 0); ledcWrite(motorB_IN2, abs(rightPWM)); } else { ledcWrite(motorB_IN1, 0); ledcWrite(motorB_IN2, 0); } } // 緊急停止関数 void stopAll() { targetLeftPWM = 0; targetRightPWM = 0; currentLeftPWM = 0; currentRightPWM = 0; setMotors(0, 0); } // BLEサーバーの接続状態コールバック class MyServerCallbacks: public BLEServerCallbacks { void onConnect(BLEServer* pServer) { deviceConnected = true; Serial.println("デバイス接続完了"); lastCommunicationTime = millis(); } void onDisconnect(BLEServer* pServer) { deviceConnected = false; Serial.println("デバイス切断"); stopAll(); // 切断時に強制停止 } }; // Webアプリからのデータ受信コールバック class MyCallbacks: public BLECharacteristicCallbacks { void onWrite(BLECharacteristic *pCharacteristic) { // Core v3.x対応: getValue()はString型 String value = pCharacteristic->getValue(); if (value.length() > 0) { lastCommunicationTime = millis(); // パース処理 (ここでは目標値を更新するだけ。実際の出力はloopで行う) int commaIndex = value.indexOf(','); if(commaIndex > 0) { targetLeftPWM = value.substring(0, commaIndex).toInt(); targetRightPWM = value.substring(commaIndex + 1).toInt(); } } } }; void setup() { // ブラウンアウト(電圧降下による再起動)対策 WRITE_PERI_REG(RTC_CNTL_BROWN_OUT_REG, 0); Serial.begin(115200); // --- モーターピンの初期化 (ESP32 Core v3対応 ledcAttach) --- ledcAttach(motorA_IN1, PWM_FREQ, PWM_RESOLUTION); ledcAttach(motorA_IN2, PWM_FREQ, PWM_RESOLUTION); ledcAttach(motorB_IN1, PWM_FREQ, PWM_RESOLUTION); ledcAttach(motorB_IN2, PWM_FREQ, PWM_RESOLUTION); stopAll(); // --- BLEの初期化 --- BLEDevice::init("ESP32_RC"); pServer = BLEDevice::createServer(); pServer->setCallbacks(new MyServerCallbacks()); BLEService *pService = pServer->createService(SERVICE_UUID); pCharacteristic = pService->createCharacteristic( CHARACTERISTIC_UUID, BLECharacteristic::PROPERTY_READ | BLECharacteristic::PROPERTY_WRITE | BLECharacteristic::PROPERTY_NOTIFY | BLECharacteristic::PROPERTY_INDICATE ); pCharacteristic->setCallbacks(new MyCallbacks()); pCharacteristic->addDescriptor(new BLE2902()); pService->start(); // アドバタイジングの開始 BLEAdvertising *pAdvertising = BLEDevice::getAdvertising(); pAdvertising->addServiceUUID(SERVICE_UUID); pAdvertising->setScanResponse(false); pAdvertising->setMinPreferred(0x0); BLEDevice::startAdvertising(); Serial.println("Webブラウザからの接続待機中..."); } void loop() { // 切断されたときの再アドバタイズ処理 if (!deviceConnected && oldDeviceConnected) { delay(500); pServer->startAdvertising(); Serial.println("再アドバタイズ開始 (接続待ち)"); oldDeviceConnected = deviceConnected; } if (deviceConnected && !oldDeviceConnected) { oldDeviceConnected = deviceConnected; } // --- フェールセーフ --- if (deviceConnected && (millis() - lastCommunicationTime > FAILSAFE_TIMEOUT)) { targetLeftPWM = 0; targetRightPWM = 0; } // ================================================================= // ★ ソフトスタート処理 (突入電流防止) ★ // 目標のPWM値(target)に向かって、現在のPWM値(current)を滑らかに追従させる // ================================================================= float smoothingFactor = 0.05; // この数値を小さくするほど滑らか(遅延)になる (例: 0.02〜0.1) currentLeftPWM += (targetLeftPWM - currentLeftPWM) * smoothingFactor; currentRightPWM += (targetRightPWM - currentRightPWM) * smoothingFactor; // モーターへ出力 // （値が非常に小さい場合は停止扱いにしてノイズを防ぐ） int outL = round(currentLeftPWM); int outR = round(currentRightPWM); if(abs(outL) < 10) outL = 0; if(abs(outR) < 10) outR = 0; setMotors(outL, outR); delay(10); // 10ms周期で徐々に出力を上げる }
